Erstelle: frontend/src/components/QuickCreate.jsx

Konzept:
  Ein Dropdown-Feld mit integriertem "+ Neu anlegen" Button.
  Wenn geklickt, öffnet sich ein Mini-Modal (schmaler als das Haupt-Modal)

  - options: Array der vorhandenen Optionen [{id, label}]
  - onCreateNew: async function(formData) → gibt neues Objekt zurück
  - createForm: JSX der Felder für das Mini-Modal
  - label: Bezeichnung des Feldes
  - placeholder: Placeholder-Text
  - disabled: boolean
  - searchable: boolean (Suche in Dropdown aktivieren)

Implementierung:
───────────────────────────────────────────────────────────────
export default function QuickCreate({
  value, onChange, options, onCreateNew,
  createForm, label, placeholder, disabled, searchable = true
}) {
  const [showCreate, setShowCreate] = useState(false);
  const [search, setSearch] = useState('');
  const [open, setOpen] = useState(false);
  const [creating, setCreating] = useState(false);

  const filtered = searchable
    ? options.filter(o => o.label.toLowerCase().includes(search.toLowerCase()))
    : options;

  const selectedLabel = options.find(o => o.id == value)?.label;

  async function handleCreate(formData) {
    setCreating(true);
    try {
      const newItem = await onCreateNew(formData);
      onChange(newItem.id);
      setShowCreate(false);
    } finally {
      setCreating(false);
    }
  }

  return (
    <div className="quick-create-wrapper">
      <label className="form-label">{label}</label>

      {/* Haupt-Select mit Suche */}
      <div className="quick-create-select" onClick={() => !disabled && setOpen(o => !o)}>
        <span className={selectedLabel ? '' : 'placeholder'}>
          {selectedLabel || placeholder || 'Auswählen...'}
        </span>
        <ChevronIcon />
      </div>

      {open && (
        <div className="quick-create-dropdown">
          {searchable && (
            <input
              className="quick-create-search"
              placeholder="Suchen..."
              value={search}
              onChange={e => setSearch(e.target.value)}
              onClick={e => e.stopPropagation()}
              autoFocus
            />
          )}
          <div className="quick-create-options">
            {filtered.map(o => (
              <div
                key={o.id}
                className={`quick-create-option ${o.id == value ? 'selected' : ''}`}
                onClick={() => { onChange(o.id); setOpen(false); setSearch(''); }}
              >
                {o.label}
                {o.sublabel && <span className="sublabel">{o.sublabel}</span>}
              </div>
            ))}
            {filtered.length === 0 && (
              <div className="quick-create-empty">
                Keine Treffer für "{search}"
              </div>
            )}
          </div>
          {/* "+ Neu anlegen" immer am Ende */}
          <div className="quick-create-new-btn" onClick={() => { setOpen(false); setShowCreate(true); }}>
            <PlusIcon /> Neu anlegen
          </div>
        </div>
      )}

      {/* Mini-Modal für neue Entität */}
      {showCreate && (
        <QuickCreateModal
          title={`Neuen ${label} anlegen`}
          onClose={() => setShowCreate(false)}
          onSubmit={handleCreate}
          creating={creating}
        >
          {createForm}
        </QuickCreateModal>
      )}
    </div>
  );
}

  function normalizeString(s) {
    return s?.toLowerCase()
             .trim()
             .replace(/[\s\-_\.]+/g, ' ')
             .normalize('NFD')
             .replace(/[\u0300-\u036f]/g, ''); // Umlaute normalisieren
  }

  function wordOverlap(a, b) {
    const wa = new Set(normalizeString(a).split(' ').filter(w => w.length > 2));
    const wb = new Set(normalizeString(b).split(' ').filter(w => w.length > 2));
    let count = 0;
    for (const w of wa) if (wb.has(w)) count++;
    return count;
  }

Erstelle: backend/src/utils/queryBuilder.js

  Baut flexible PostgreSQL WHERE + ORDER BY Clauses aus URL-Parametern:

  module.exports = function buildQuery(params, allowedFilters, allowedSorts) {
    const where = [];
    const values = [];
    let i = 1;

    for (const [key, config] of Object.entries(allowedFilters)) {
      const val = params[key];
      if (!val) continue;

      switch (config.type) {
        case 'exact':
          where.push(`${config.col} = $${i++}`);
          values.push(val);
          break;
        case 'ilike':
          where.push(`${config.col} ILIKE $${i++}`);
          values.push(`%${val}%`);
          break;
        case 'in':
          // val = "1,2,3" → IN ($1,$2,$3)
          const ids = val.split(',').map(v => v.trim()).filter(Boolean);
          if (ids.length) {
            where.push(`${config.col} = ANY($${i++}::int[])`);
            values.push(ids);
          }
          break;
        case 'boolean':
          where.push(`${config.col} = $${i++}`);
          values.push(val === 'true');
          break;
        case 'date_from':
          where.push(`${config.col} >= $${i++}`);
          values.push(val);
          break;
        case 'date_to':
          where.push(`${config.col} <= $${i++}`);
          values.push(val);
          break;
        case 'gte':
          where.push(`${config.col} >= $${i++}`);
          values.push(parseInt(val));
          break;
        case 'lte':
          where.push(`${config.col} <= $${i++}`);
          values.push(parseInt(val));
          break;
      }
    }

    // Sortierung
    const sortField = params.sort_by;
    const sortDir = params.sort_dir === 'desc' ? 'DESC' : 'ASC';
    const orderBy = (sortField && allowedSorts[sortField])
      ? `ORDER BY ${allowedSorts[sortField]} ${sortDir}`
      : `ORDER BY ${allowedSorts['default']} DESC`;

    // Pagination
    const limit = Math.min(parseInt(params.limit) || 50, 200);
    const offset = parseInt(params.offset) || 0;

    return {
      whereClause: where.length ? 'WHERE ' + where.join(' AND ') : '',
      values,
      orderBy,
      pagination: `LIMIT $${i++} OFFSET $${i++}`,
      paginationValues: [limit, offset],
      nextIndex: i,
    };
  };

AUFGABE 4.3 — Frontend: useFilter Hook
────────────────────────────────────────
Erstelle: frontend/src/hooks/useFilter.js

  // Zentraler Hook für Filter/Sort/Pagination State
  export function useFilter(defaultSort = 'created_at', defaultDir = 'desc') {
    const [filters, setFilters] = useState({});
    const [sort, setSort] = useState({ by: defaultSort, dir: defaultDir });
    const [page, setPage] = useState({ limit: 50, offset: 0 });

    function setFilter(key, value) {
      setFilters(f => ({ ...f, [key]: value || undefined }));
      setPage(p => ({ ...p, offset: 0 })); // Reset auf Seite 1
    }

    function toggleSort(field) {
      setSort(s => ({
        by: field,
        dir: s.by === field && s.dir === 'asc' ? 'desc' : 'asc',
      }));
    }

    function clearFilters() {
      setFilters({});
      setPage({ limit: 50, offset: 0 });
    }

    const queryParams = {
      ...filters,
      sort_by: sort.by,
      sort_dir: sort.dir,
      limit: page.limit,
      offset: page.offset,
    };

    return { filters, sort, page, setFilter, toggleSort, setPage, clearFilters, queryParams };
  }
